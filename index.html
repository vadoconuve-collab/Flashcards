<div class="container" id="setup">

    <h2>📚 Mis Flashcards</h2>

    <p>Sube tu archivo .csv (Formato: Pregunta, Respuesta)</p>

    <input type="file" id="csvFile" accept=".csv" style="margin-bottom: 20px;">

    <div id="menu" class="hidden">

        <button class="btn" onclick="iniciarTest()">Empezar Test Multi-opción</button>

    </div>

</div>


<div class="container hidden" id="test-area">

    <div id="progreso"></div>

    <div class="card" id="pregunta-texto"></div>

    <div id="opciones-container"></div>

</div>


<div class="container hidden" id="resultados">

    <h2>Resultados</h2>

    <h3 id="puntuacion-final"></h3>

    <div id="lista-errores"></div>

    <button class="btn" onclick="location.reload()">Reiniciar</button>

</div>


<script>

    let mazo = [];

    let indiceActual = 0;

    let puntuacion = 0;

    let errores = [];


    // 1. Cargar CSV

    document.getElementById('csvFile').addEventListener('change', function(e) {

        const file = e.target.files[0];

        Papa.parse(file, {

            complete: function(results) {

                mazo = results.data.filter(f => f.length >= 2).map(f => ({ p: f[0], r: f[1] }));

                if(mazo.length > 0) document.getElementById('menu').classList.remove('hidden');

            }

        });

    });


    function iniciarTest() {

        indiceActual = 0;

        puntuacion = 0;

        errores = [];

        document.getElementById('setup').classList.add('hidden');

        document.getElementById('test-area').classList.remove('hidden');

        mostrarSiguientePregunta();

    }


    function mostrarSiguientePregunta() {

        if (indiceActual >= mazo.length) return mostrarResultados();


        const data = mazo[indiceActual];

        document.getElementById('progreso').innerText = `Pregunta ${indiceActual + 1} de ${mazo.length}`;

        document.getElementById('pregunta-texto').innerText = data.p;

        

        // Generar opciones (Correcta + 3 aleatorias)

        let opciones = [data.r];

        let otrasRespuestas = mazo.filter(t => t.r !== data.r).map(t => t.r);

        while(opciones.length < Math.min(4, mazo.length)) {

            let randomRes = otrasRespuestas[Math.floor(Math.random() * otrasRespuestas.length)];

            if(!opciones.includes(randomRes)) opciones.push(randomRes);

        }

        opciones.sort(() => Math.random() - 0.5);


        const container = document.getElementById('opciones-container');

        container.innerHTML = '';

        opciones.forEach(opt => {

            const btn = document.createElement('button');

            btn.className = 'btn option-btn';

            btn.innerText = opt;

            btn.onclick = () => verificarRespuesta(opt);

            container.appendChild(btn);

        });

    }


    function verificarRespuesta(elegida) {

        const correcta = mazo[indiceActual].r;

        if (elegida === correcta) {

            puntuacion++;

        } else {

            errores.push({ preg: mazo[indiceActual].p, corr: correcta, tuya: elegida });

        }

        indiceActual++;

        mostrarSiguientePregunta();

    }


    function mostrarResultados() {

        document.getElementById('test-area').classList.add('hidden');

        document.getElementById('resultados').classList.remove('hidden');

        document.getElementById('puntuacion-final').innerText = `Puntuación: ${puntuacion} / ${mazo.length}`;

        

        const lista = document.getElementById('lista-errores');

        lista.innerHTML = errores.length > 0 ? '<h4>Repasa tus fallos:</h4>' : '<h4>¡Perfecto! No has fallado nada.</h4>';

        

        errores.forEach(e => {

            lista.innerHTML += `

                <div class="result-item">

                    <strong>P: ${e.preg}</strong><br>

                    Tu respuesta: <span class="err-text">${e.tuya}</span><br>

                    Correcta: <span class="corr-text">${e.corr}</span>

                </div>`;

        });

    }

</script>
